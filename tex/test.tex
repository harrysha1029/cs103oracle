\begin{definition}[Metric Space]
% tags: wefoijawefaoweifj
A set $X$ and function $d$ is a metric space if for all $x, y, z \in X$ the following hold:
    \begin{enumerate}
        \item $d(x, x) = 0$
        \item $d(x, y) = d(y, x)$
        \item $d(x, z) \leq d(x, y) + d(y, z)$
    \end{enumerate}
\end{definition}
